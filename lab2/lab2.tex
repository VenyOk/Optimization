using Plots

function svenn(f, x0, h)
    x1 = x0
    x2 = x0 + h
    if f(x2) > f(x1)
        h = -h
        x2 = x0 + h
    end
    while f(x2) < f(x1)
        x1 = x2
        x2 = x1 + h
        h = 2 * h
    end
    a = min(x1, x2)
    b = max(x1, x2)
    return a, b
end

function fib(n)
    if n <= 1
        return n
    end
    a, b = 0, 1
    for _ in 2:n
        a, b = b, a + b
    end
    return b
end

function fibonacci_search(f, a, b, eps)
    n = 1
    while fib(n + 2) < (b - a) / eps
        n += 1
    end
    x_seq = Float64[]
    y_seq = Float64[]
    k = 1
    x1 = a + (b - a) * fib(n) / fib(n + 2)
    x2 = a + (b - a) * fib(n + 1) / fib(n + 2)
    f1 = f(x1)
    f2 = f(x2)
    push!(x_seq, x1)
    push!(y_seq, f1)
    push!(x_seq, x2)
    push!(y_seq, f2)
    while k < n
        if f1 > f2
            a = x1
            x1 = x2
            f1 = f2
            x2 = a + (b - a) * fib(n - k + 1) / fib(n - k + 2)
            f2 = f(x2)
        else
            b = x2
            x2 = x1
            f2 = f1
            x1 = a + (b - a) * fib(n - k) / fib(n - k + 2)
            f1 = f(x1)
        end
        push!(x_seq, f1 < f2 ? x1 : x2)
        push!(y_seq, f1 < f2 ? f1 : f2)
        k += 1
    end
    candidates = [a, b, (a + b) / 2, x1, x2]
    x_min = candidates[1]
    for c in candidates
        if f(c) < f(x_min)
            x_min = c
        end
    end
    push!(x_seq, x_min)
    push!(y_seq, f(x_min))
    return x_min, x_seq, y_seq, n
end

function reverse_variable_step(f, a, b, eps)
    x_seq = Float64[]
    y_seq = Float64[]
    x = a
    h = (b - a) * 0.1
    if h < eps
        h = eps * 2
    end
    push!(x_seq, x)
    push!(y_seq, f(x))
    iters = 0
    while abs(h) > eps
        x_cand = x + h
        if x_cand < a
            x_cand = a
        end
        if x_cand > b
            x_cand = b
        end
        if f(x_cand) < f(x)
            x = x_cand
            push!(x_seq, x)
            push!(y_seq, f(x))
            h = 2 * h
            if x + h > b && h > 0
                h = (b - x) * 0.5
            end
            if x + h < a && h < 0
                h = (a - x) * 0.5
            end
        else
            h = -h / 2
        end
        iters += 1
    end
    if length(x_seq) == 0 || x_seq[end] != x
        push!(x_seq, x)
        push!(y_seq, f(x))
    end
    return x, x_seq, y_seq, iters
end

function parabola_value(x1, f1, x2, f2, x3, f3, x)
    d12 = (x1 - x2) * (x1 - x3)
    d21 = (x2 - x1) * (x2 - x3)
    d31 = (x3 - x1) * (x3 - x2)
    (abs(d12) < 1e-14 || abs(d21) < 1e-14 || abs(d31) < 1e-14) && return f2
    f1 * (x - x2) * (x - x3) / d12 + f2 * (x - x1) * (x - x3) / d21 + f3 * (x - x1) * (x - x2) / d31
end

function powell_search(f, a, b, eps)
    x_seq = Float64[]
    y_seq = Float64[]
    parabolas = Tuple{Float64,Float64,Float64,Float64,Float64,Float64}[]
    x1 = a
    x3 = b
    x2 = (a + b) / 2
    f1 = f(x1)
    f2 = f(x2)
    f3 = f(x3)
    if f2 > f1 && f2 > f3
        x2 = f1 < f3 ? x1 + (x3 - x1) * 0.382 : x1 + (x3 - x1) * 0.618
        f2 = f(x2)
    end
    push!(x_seq, x1)
    push!(y_seq, f1)
    push!(x_seq, x2)
    push!(y_seq, f2)
    push!(x_seq, x3)
    push!(y_seq, f3)
    iters = 0
    while (x3 - x1) > eps
        push!(parabolas, (x1, f1, x2, f2, x3, f3))
        denom = (x2 - x3) * f1 + (x3 - x1) * f2 + (x1 - x2) * f3
        if abs(denom) < 1e-14
            break
        end
        x_new = 0.5 * ((x2^2 - x3^2) * f1 + (x3^2 - x1^2) * f2 + (x1^2 - x2^2) * f3) / denom
        if x_new <= x1 || x_new >= x3
            x_new = (x1 + x2) / 2
        end
        f_new = f(x_new)
        push!(x_seq, x_new)
        push!(y_seq, f_new)
        idx_worst = argmax([f1, f2, f3])
        if idx_worst == 1
            x1 = x_new
            f1 = f_new
        elseif idx_worst == 2
            x2 = x_new
            f2 = f_new
        else
            x3 = x_new
            f3 = f_new
        end
        ord = sortperm([x1, x2, x3])
        xs = [x1, x2, x3][ord]
        fs = [f1, f2, f3][ord]
        x1, x2, x3 = xs[1], xs[2], xs[3]
        f1, f2, f3 = fs[1], fs[2], fs[3]
        iters += 1
    end
    x_min = f1 < f2 && f1 < f3 ? x1 : (f2 < f3 ? x2 : x3)
    push!(x_seq, x_min)
    push!(y_seq, f(x_min))
    return x_min, x_seq, y_seq, iters, parabolas
end

function num_derivative(f, x, delta=1e-7)
    return (f(x + delta) - f(x - delta)) / (2 * delta)
end

function num_second_derivative(f, x, delta=1e-6)
    return (f(x + delta) - 2 * f(x) + f(x - delta)) / (delta^2)
end

function prove_unimodality(f, a, b, n_points=100)
    step = (b - a) / (n_points - 1)
    sign_changes = 0
    prev_der = num_derivative(f, a)
    for i in 2:n_points
        x = a + (i - 1) * step
        der = num_derivative(f, x)
        if (prev_der < 0 && der >= 0) || (prev_der > 0 && der <= 0)
            sign_changes += 1
        end
        prev_der = der
    end
    unimodal = (sign_changes == 1)
    println("Унимодальность на [", a, ", ", b, "]: ", unimodal)
    return unimodal
end

function prove_minimum_rule(f, x_min, precision, a=-Inf, b=Inf)
    delta = max(1e-10, precision * 0.5)
    f0 = f(x_min)
    tol = max(1e-9, 1e-6 * max(abs(f0), 1.0))
    check_left = x_min - delta >= a
    check_right = x_min + delta <= b
    rain_left = !check_left || (f(x_min - delta) >= f0 - tol)
    rain_right = !check_right || (f(x_min + delta) >= f0 - tol)
    is_min_rain = rain_left && rain_right
    println("Правило дождя: при h=", delta, " : ", rain_left, ", ", rain_right)
    println("Точка является минимумом: ", is_min_rain)
    return is_min_rain
end

function local_lims(xseq, yseq)
    xlo, xhi = minimum(xseq), maximum(xseq)
    ylo, yhi = minimum(yseq), maximum(yseq)
    span_x = max(xhi - xlo, min_span_x)
    span_y = max(yhi - ylo, max(abs(ylo), abs(yhi), 1e-8) * 0.08)
    dx = span_x * margin
    dy = span_y * margin
    (xlo - dx, xhi + dx, ylo - dy, yhi + dy)
end
function plot_path_on_curve!(plt, f, xseq, color, marker, label; n_pts=50)
    for i in 1:length(xseq)-1
        x_seg = range(xseq[i], xseq[i+1], length=n_pts)
        plot!(plt, x_seg, f.(x_seg), color=color, lw=1.2, label=(i == 1 ? label : ""))
    end
    plot!(plt, xseq, f.(xseq), color=color, lw=0, marker=marker, ms=6, label="")
end
